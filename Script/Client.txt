CREATE TABLE Client (
Nit varchar(255) NOT NULL,
Name varchar(255) NOT NULL,
PRIMARY KEY (Nit));

CREATE PROCEDURE JobTime_Get_Client
AS
BEGIN
  SELECT * FROM dbo.Client; 
END

Exec dbo.JobTime_Get_Client

CREATE PROCEDURE JobTime_Post_Client
@Nit varchar(255),
@Name varchar(255)
AS
BEGIN	
	IF EXISTS(SELECT 1 FROM dbo.Client WHERE Nit = @Nit)
	BEGIN
	    UPDATE dbo.Client
		SET Name = @Name
		WHERE Nit = @Nit;
	END
	ELSE
	BEGIN
		INSERT dbo.Client
		( Nit, Name)
		VALUES
		( @Nit, @Name)
	END
END

CREATE PROCEDURE JobTime_Delete_Client
@Nit varchar(255)
AS
BEGIN
	SET DATEFORMAT DMY;
	SET LANGUAGE SPANISH;
	DECLARE @MsgError NVARCHAR(MAX) = '';	
	BEGIN TRY
	    IF NOT EXISTS(SELECT 1 FROM dbo.Client WHERE Nit = @Nit)
		BEGIN
			SET @MsgError = 'The Client with Nit: ' + CONVERT(VARCHAR(10), @Nit) + ' does not exist.';
			RAISERROR(@MsgError, 16, 217);
		END
		ELSE
		BEGIN
			DELETE dbo.Client WHERE Nit = @Nit	    
		END
	END TRY
	BEGIN CATCH
	    SET @MsgError = ERROR_MESSAGE();
		RAISERROR(@MsgError, 16, 217);
	END CATCH
END
