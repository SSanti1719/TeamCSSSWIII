CREATE TABLE Manager (
Nit varchar(255) NOT NULL,
Name varchar(255) NOT NULL,
Email varchar(255) NOT NULL,
JobTitle varchar(255) NOT NULL,
PRIMARY KEY (Nit));

CREATE PROCEDURE JobTime_Get_Manager
AS
BEGIN
  SELECT * FROM dbo.Manager; 
END

Exec dbo.JobTime_Get_Manager

CREATE PROCEDURE JobTime_Post_Manager
@Nit varchar(255),
@Name varchar(255),
@Email varchar(255),
@JobTitle varchar(255)
AS
BEGIN	
	IF EXISTS(SELECT 1 FROM dbo.Manager WHERE Nit = @Nit)
	BEGIN
	    UPDATE dbo.Manager
		SET Name = @Name,
			Email = @Email,
			JobTitle = @JobTitle
		WHERE Nit = @Nit;
	END
	ELSE
	BEGIN
		INSERT dbo.Manager
		( Nit, Name, Email, JobTitle)
		VALUES
		( @Nit, @Name, @Email, @JobTitle )
	END
END

CREATE PROCEDURE JobTime_Delete_Manager
@Nit varchar(255)
AS
BEGIN
	SET DATEFORMAT DMY;
	SET LANGUAGE SPANISH;
	DECLARE @MsgError NVARCHAR(MAX) = '';	
	BEGIN TRY
	    IF NOT EXISTS(SELECT 1 FROM dbo.Manager WHERE Nit = @Nit)
		BEGIN
			SET @MsgError = 'The employee with Nit: ' + CONVERT(VARCHAR(10), @Nit) + ' does not exist.';
			RAISERROR(@MsgError, 16, 217);
		END
		ELSE
		BEGIN
			DELETE dbo.Manager WHERE Nit = @Nit	    
		END
	END TRY
	BEGIN CATCH
	    SET @MsgError = ERROR_MESSAGE();
		RAISERROR(@MsgError, 16, 217);
	END CATCH
END
