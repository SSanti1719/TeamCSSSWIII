CREATE TABLE Report (
Id Int NOT NULL,
IdAssign Int NOT NULL,
PRIMARY KEY (Id),
FOREIGN KEY (IdAssign) REFERENCES dbo.Assign(Id));

CREATE PROCEDURE JobTime_Get_Report
AS
BEGIN
  SELECT * FROM dbo.Report; 
END

Exec dbo.JobTime_Get_Report


CREATE PROCEDURE JobTime_Post_Report
@Id INT,
@IdAssign INT
AS
BEGIN	
	IF EXISTS(SELECT 1 FROM dbo.Assign WHERE Id = @Id)
	BEGIN
	    UPDATE dbo.Report
		SET IdAssign = @IdAssign
		WHERE Id = @Id;
	END
	ELSE
	BEGIN
	    SELECT @Id = ISNULL(MAX(Id), 0) + 1 FROM dbo.Report;
		INSERT dbo.Report
		( Id, IdAssign )
		VALUES
		( @Id, @IdAssign)
	END
END

CREATE PROCEDURE JobTime_Delete_Report
@Id varchar(255)
AS
BEGIN
	SET DATEFORMAT DMY;
	SET LANGUAGE SPANISH;
	DECLARE @MsgError NVARCHAR(MAX) = '';	
	BEGIN TRY
	    IF NOT EXISTS(SELECT 1 FROM dbo.Report WHERE Id = @Id)
		BEGIN
			SET @MsgError = 'The Assign with Id: ' + CONVERT(VARCHAR(10), @Id) + ' does not exist.';
			RAISERROR(@MsgError, 16, 217);
		END
		ELSE
		BEGIN
			DELETE dbo.Report WHERE Id = @Id	    
		END
	END TRY
	BEGIN CATCH
	    SET @MsgError = ERROR_MESSAGE();
		RAISERROR(@MsgError, 16, 217);
	END CATCH
END
